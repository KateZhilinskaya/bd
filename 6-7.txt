 db.unicorns.update({name : "Ayna"}, {$inc: { vampires: 5}})
----------------с помощью $inc увеличиваем число вампиров у Ayna на 5
db.unicorns.update({name : "Kenny"}, {$unset: {vampires: 1}})
----удалили у Kenny поле vampires
db.unicorns.update({name : "Leia"}, {$push: {loves: "melon"}})
---с помощью $push добавили еще одно значение в массив Loves
---Оператор $addToSet подобно оператору $push добавляет объекты в массив. 
db.unicorns.update({name : "Pilot"},{$addToSet: {loves: {$each: ["melon", "carrot", "pumpkin"]}}})
---с помощью $each можем добавить несколько элементов в массив
db.unicorns.update({name : "Nimue"}, {$pop: {loves: 1}})
-----с помощью $pop удаляем элемент из массива , если указать 1-то удалиться первый элемент с конца, если указать -1 -удалится первый элемент с начала 
db.unicorns.update({name : "Pilot"}, {$pull: {loves: "carrot"}})
-----Оператор $pull удаляет каждое вхождение элемента в массив. Например, через оператор $push можно добавить одно и то же значение в массив несколько раз. Далее с помощью $pull удалить его
----Если нужно удалить не одно значение, а сразу несколько, тогда можно применить оператор $pullAll:
